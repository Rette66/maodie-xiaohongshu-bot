```
猫的作息 → 账号发布时间：

白天（8:00-18:00）：睡觉
  → 不发内容，不回复

傍晚（18:00-23:00）：醒来，观察
  → 偶尔发一条

深夜（23:00-4:00）：活跃期
  → 主要发布时间
  → 突然发疯时间

凌晨（4:00-8:00）：继续睡
```

```
猫的逻辑：想发就发，不想发就不发
```

### Agent决策逻辑

```python
class CatLogicAgent:
    """基于猫的逻辑的Agent"""
    
    def should_reply(self, comment) -> bool:
        """猫要不要回复？"""
        # 猫的逻辑：大部分时候不理你
        if random.random() > 0.1:
            return False
        
        # 深夜更容易回复（猫醒了）
        if self.is_night():
            return random.random() < 0.2
        
        return random.random() < 0.05
    
    def get_reply(self, comment) -> str:
        """猫怎么回复？"""
        replies = ["哈", "哈哈", "哈哈哈"]
        return random.choice(replies)
    
    def should_post(self) -> bool:
        """猫要不要发内容？"""
        hour = datetime.now().hour
        
        # 白天睡觉
        if 8 <= hour < 18:
            return random.random() < 0.05
        
        # 深夜活跃
        if 23 <= hour or hour < 4:
            return random.random() < 0.3
        
        # 其他时间
        return random.random() < 0.1
    
    def get_content(self) -> dict:
        """猫发什么内容？"""
        content_types = [
            ("表情包", 0.8),
            ("纯文字", 0.1),
            ("突然发疯", 0.1),
        ]
        
        # 随机选择内容类型
        content_type = random.choices(
            [c[0] for c in content_types],
            weights=[c[1] for c in content_types]
        )[0]
        
        return {
            "type": content_type,
            "text": self._generate_cat_text(content_type),
            "tags": self._get_random_tags(),
        }
    
    def _generate_cat_text(self, content_type) -> str:
        """生成猫的文字"""
        if content_type == "表情包":
            return ""  # 纯表情包不配文
        elif content_type == "纯文字":
            return random.choice(["哈", "哈哈", "哈哈哈"])
        else:  # 突然发疯
            return "哈" * random.randint(5, 20)
    
    def _get_random_tags(self) -> list:
        """随机标签"""
        if random.random() < 0.3:
            return []  # 30%概率不带标签
        return random.sample(["耄耋", "圆头耄耋", "哈人", "抽象"], k=random.randint(1, 2))
```
